317%LQG Postion Estimator and Controller
318% Observer:
319%        x[n|n]   = x[n|n-1] + M(y[n] - Cx[n|n-1] - Du[n])
320%        x[n+1|n] = Ax[n|n] + Bu[n]
321%
322% $Author: Tobias Naegeli $    $Date: 2014 $    $Revision: 3 $
323%
324%
325% Arguments:
326% approx_prediction: if 1 then the exponential map is approximated with a

330% use_inertia_matrix: set to true if you have the inertia matrix J for your
331% quadrotor
332% xa_apo_k: old state vectotr
333% zFlag: if sensor measurement is available [gyro, acc, mag]

335% z: measurements [gyro, acc, mag]
336% q_rotSpeed: process noise gyro
337% q_rotAcc: process noise gyro acceleration
338% q_acc: process noise acceleration
339% q_mag: process noise magnetometer
340% r_gyro: measurement noise gyro
341% r_accel: measurement noise accel
342% r_mag: measurement noise mag
343% J: moment of inertia matrix

345

404% compute the apriori state estimate from the previous aposteriori estimate

407    wak =[wax;way;waz]+Ji*(-cross([wax;way;waz],J*[wax;way;waz]))*dt;
408else
409    wak =[wax;way;waz];
410end
411
412%body angular rates
413wk =[wx;  wy; wz] + dt*wak;
414
415%derivative of the prediction rotation matrix
416O=[0,-wz,wy;wz,0,-wx;-wy,wx,0]';
417
418%prediction of the earth z vector
419if (approx_prediction==1)
420    %e^(Odt)=I+dt*O+dt^2/2!O^2
421    % so we do a first order approximation of the exponential map
422    zek =(O*dt+single(eye(3)))*[zex;zey;zez];
423    
424else
425    zek =(single(eye(3))+O*dt+dt^2/2*O^2)*[zex;zey;zez];
426    %zek =expm2(O*dt)*[zex;zey;zez]; not working because use double
427    %precision
428end

430
431
432%prediction of the magnetic vector
433if (approx_prediction==1)
434    %e^(Odt)=I+dt*O+dt^2/2!O^2
435    % so we do a first order approximation of the exponential map
436    muk =(O*dt+single(eye(3)))*[mux;muy;muz];
437else
438     muk =(single(eye(3))+O*dt+dt^2/2*O^2)*[mux;muy;muz];
439    %muk =expm2(O*dt)*[mux;muy;muz]; not working because use double
440    %precision
441end
442
443x_apr=[wk;wak;zek;muk];
444
445% compute the apriori error covariance estimate from the previous
446%aposteriori estimate
447
448EZ=[0,zez,-zey;
449    -zez,0,zex;
450    zey,-zex,0]';
451MA=[0,muz,-muy;
452    -muz,0,mux;
453    muy,-mux,0]';
454
455E=single(eye(3));
456Z=single(zeros(3));
457
458A_lin=[ Z,  E,  Z,  Z
459    Z,  Z,  Z,  Z
460    EZ, Z,  O,  Z
461    MA, Z,  Z,  O];
462
463A_lin=eye(12)+A_lin*dt;
464
465%process covariance matrix
466
467persistent Q
468if (isempty(Q))
469    Q=diag([ q_rotSpeed,q_rotSpeed,q_rotSpeed,...
470        q_rotAcc,q_rotAcc,q_rotAcc,...
471        q_acc,q_acc,q_acc,...
472        q_mag,q_mag,q_mag]);
473end
474
475P_apr=A_lin*P_apo*A_lin'+Q;

477
478%% update
479if zFlag(1)==1&&zFlag(2)==1&&zFlag(3)==1

481%     R=[r_gyro,0,0,0,0,0,0,0,0;

483%         0,0,r_gyro,0,0,0,0,0,0;
484%         0,0,0,r_accel,0,0,0,0,0;
485%         0,0,0,0,r_accel,0,0,0,0;
486%         0,0,0,0,0,r_accel,0,0,0;
487%         0,0,0,0,0,0,r_mag,0,0;
488%         0,0,0,0,0,0,0,r_mag,0;
489%         0,0,0,0,0,0,0,0,r_mag];
490     R_v=[r_gyro,r_gyro,r_gyro,r_accel,r_accel,r_accel,r_mag,r_mag,r_mag];
491    %observation matrix
492    %[zw;ze;zmk];
493    H_k=[  E,     Z,      Z,    Z;
494        Z,     Z,      E,    Z;
495        Z,     Z,      Z,    E];
496    
497    y_k=z(1:9)-H_k*x_apr;
498    
499    
500    %S_k=H_k*P_apr*H_k'+R;
501     S_k=H_k*P_apr*H_k';
502     S_k(1:9+1:end) = S_k(1:9+1:end) + R_v;
503    K_k=(P_apr*H_k'/(S_k));
504    
505    
506    x_apo=x_apr+K_k*y_k;
507    P_apo=(eye(12)-K_k*H_k)*P_apr;
508else
509    if zFlag(1)==1&&zFlag(2)==0&&zFlag(3)==0
510        
511        R=[r_gyro,0,0;
512            0,r_gyro,0;
513            0,0,r_gyro];
514        R_v=[r_gyro,r_gyro,r_gyro];
515        %observation matrix
516        
517        H_k=[  E,     Z,      Z,    Z];
518        
519        y_k=z(1:3)-H_k(1:3,1:12)*x_apr;
520        
521       % S_k=H_k(1:3,1:12)*P_apr*H_k(1:3,1:12)'+R(1:3,1:3);
522        S_k=H_k(1:3,1:12)*P_apr*H_k(1:3,1:12)';
523        S_k(1:3+1:end) = S_k(1:3+1:end) + R_v;
524        K_k=(P_apr*H_k(1:3,1:12)'/(S_k));
525        
526        
527        x_apo=x_apr+K_k*y_k;
528        P_apo=(eye(12)-K_k*H_k(1:3,1:12))*P_apr;
529    else
530        if  zFlag(1)==1&&zFlag(2)==1&&zFlag(3)==0
531            
532%             R=[r_gyro,0,0,0,0,0;
533%                 0,r_gyro,0,0,0,0;
534%                 0,0,r_gyro,0,0,0;
535%                 0,0,0,r_accel,0,0;
536%                 0,0,0,0,r_accel,0;
537%                 0,0,0,0,0,r_accel];
